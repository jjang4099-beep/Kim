### 3-4. DB 기반 피드 생성 함수 추가

**`server.js`** — generateDailyFeed() 내 분기 추가

```js
// liber_classic
case 'liber_classic': {
  const quote = await ClassicQuote.findOneAndUpdate(
    { used: false },
    { used: true, usedAt: new Date() },
    { sort: { priority: -1, createdAt: 1 }, new: true }
  );
  if (!quote) {
    // 전부 소진 시 리셋
    await ClassicQuote.updateMany({}, { used: false });
    return generateFeedItem('liber_classic');
  }
  return {
    subId:   'liber_classic',
    title:   `📖 오늘의 고전 — ${quote.book}`,
    book:    quote.book,
    author:  quote.author,
    source:  quote.source,
    quote:   quote.quote,
    context: quote.context,
    theme:   quote.theme,
    tags:    quote.tags,
  };
}

// insight_daily
case 'insight_daily': {
  const today     = new Date().getDay(); // 0=일 ~ 6=토
  const insight   = await DailyInsight.findOneAndUpdate(
    { used: false, dayOfWeek: today },
    { used: true, usedAt: new Date() },
    { sort: { createdAt: 1 }, new: true }
  );
  if (!insight) {
    await DailyInsight.updateMany({ dayOfWeek: today }, { used: false });
    return generateFeedItem('insight_daily');
  }
  return {
    subId:      'insight_daily',
    title:      `${insight.icon} ${insight.label} — ${insight.topic}`,
    topic:      insight.topic,
    headline:   insight.headline,
    body:       insight.body,
    realLife:   insight.realLife,
    question:   insight.question,
    tags:       insight.tags,
    icon:       insight.icon,
    color:      insight.color,
  };
}

// idiom_daily — DB 기반으로 전환
case 'idiom_daily': {
  const idiom = await IdiomCard.findOneAndUpdate(
    { used: false },
    { used: true, usedAt: new Date() },
    { sort: { createdAt: 1 }, new: true }
  );
  if (!idiom) {
    await IdiomCard.updateMany({}, { used: false });
    return generateFeedItem('idiom_daily');
  }
  return {
    subId:     'idiom_daily',
    title:     `📜 오늘의 고사성어 — ${idiom.idiom}`,
    idiom:     idiom.idiom,
    hanja:     idiom.hanja,
    meaning:   idiom.meaning,
    origin:    idiom.origin,
    example:   idiom.example,
    modernUse: idiom.modernUse,
    tags:      idiom.tags,
  };
}

## ⚠️ 전체 공통 주의 사항

```
1. 기존 Item 데이터 보존
   contentType 기본값 'knowledge' → 기존 아이템 전부 자동 호환

2. DB 스크립트는 로컬에서만 실행
   프로덕션 서버에서 직접 실행 금지

3. quote_daily 삭제 시 기존 저장 아이템은 그대로 유지
   새로 생성만 안 하면 됨 (DB 데이터 삭제 불필요)

4. CLAUDE.md 규칙 항상 준수
   전역 변수 금지 / fetchJSON 사용 / CSS 변수 사용 / toISOString 금지

5. 라이프 카드는 AI 분석 없음
   토큰 비용 0원 — 그냥 저장

6. 결산 AI 총평은 캐시 필수
   같은 기간 두 번 요청해도 Claude API 1회만 호출

7. 수험생 모드는 기존 직장인 모드 데이터에 영향 없음
